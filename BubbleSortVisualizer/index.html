def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

scores = [5, 1, 4, 2, 8, 3, 7, 6, 10, 9]
sorted_scores = bubble_sort(scores)
print("Sorted scores:", sorted_scores)
